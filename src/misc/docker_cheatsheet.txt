networks:
    docker network ls --> para ver las redes
    docker network rm {nombre red} --> Borrar red
    docker network create --subnet={192.168.1.0/24 (por ejemplo)} {nombre}


run:
    docker run --name {nombre} -p {puertoLocal:puertoExterior} -ip {ip} --network {network name} {nombre imagen} --> Crea el contenedor

    docker run --name yo -p 5501:5500 --ip 192.168.2.3 --network autogenerated_network imagen_socket
    
    docker start {nombre} --> Despierta un contenedor creado
    # Build manager_image
    docker build -f manager_dockerfile.dockerfile -t manager_image .
